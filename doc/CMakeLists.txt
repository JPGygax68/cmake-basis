# ============================================================================
# ChangeLog
# ============================================================================

basis_add_doc (
  ChangeLog
  GENERATOR        svn2cl
  LINELEN          79
  BREAK_BEFORE_MSG 2
  GROUP_BY_DAY SEPARATE_DAYLOGS INCLUDE_ACTIONS REPARAGRAPH
)

# ============================================================================
# API documentation
# ============================================================================

# The following CMake code adds an API documentation to the 'doc' target which
# is automatically generated by Doxygen from the in-source code comments.
# Please refer to the documentation of basis_add_doc () for details.
#
# See basis_add_doc()

basis_add_doc (
  api
  GENERATOR        Doxygen
  INPUT            "${BINARY_CODE_DIR}"
                   "${PROJECT_CODE_DIR}"
                   "${BINARY_INCLUDE_DIR}"
                   "${PROJECT_INCLUDE_DIR}"
                   # additional templates provided by BASIS
                   "${PROJECT_CODE_DIR}/cmake/ScriptConfig.cmake.in"
                   # additional Doxygen pages including main page
                   "${CMAKE_CURRENT_SOURCE_DIR}/DoxygenPages.h"
  EXCLUDE_PATTERNS # external libraries packaged with BASIS
                   "*/tclap/*"            # TCLAP
                   "*/gmock/*"            # Google Mock
                   "*/gtest/*"            # Google Test
                   "*/argh/*"             # Python Argh
                   "*/test.cxx"           # Google Mock and Test
                   "*/test_main.cxx"      # main() of unit tests based on GMock
                   # files in binary source tree
                   "cmake_install.cmake"
                   "CTestTestfile.cmake"
)

# ============================================================================
# manuals
# ============================================================================

# Add documentation files in the current directory and/or install these files
# using the macro basis_add_doc ().
#
# See basis_add_doc()

basis_add_doc (tclap)
