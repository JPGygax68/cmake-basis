25 # ----------------------------------------------------------------------------
26 ## @brief Check whether the BASIS utilities are used within a given source file.
27 #
28 # This function matches the source code against specific import patterns which
29 # are all valid imports of the BASIS utilities for the respective programming
30 # language of the specified file. If the BASIS utilities are used within the
31 # specified source file, the variable named @p VAR is set to @c TRUE.
32 # Otherwise, it is set to @c FALSE.
33 #
34 # @param [out] VAR Whether the BASIS utilites are used.
35 # @param [in] SOURCE_FILE Path of source file to check.
36 # @param [in] ARGN Source code language. If not specified, the
37 # programming language is determined automatically.
38 function (basis_utilities_check VAR SOURCE_FILE)
39  set (UTILITIES_USED FALSE)
40  if (ARGC EQUAL 2)
41  basis_get_source_language (LANGUAGE "${SOURCE_FILE}")
42  elseif (ARGC EQUAL 3)
43  set (LANGUAGE "${ARGN}")
44  else ()
45  message (FATAL_ERROR "Too many arguments given!")
46  endif ()
47  # --------------------------------------------------------------------------
48  # make file path absolute and append .in suffix if necessary
49  get_filename_component (SOURCE_FILE "${SOURCE_FILE}" ABSOLUTE)
50  if (NOT EXISTS "${SOURCE_FILE}" AND NOT SOURCE_FILE MATCHES "\\.in$" AND EXISTS "${SOURCE_FILE}.in")
51  set (SOURCE_FILE "${SOURCE_FILE}.in")
52  endif ()
53  # --------------------------------------------------------------------------
54  # C++
55  if (LANGUAGE MATCHES "CXX")
56  # read script file
57  file (READ "${SOURCE_FILE}" SOURCE)
58  # match use/require statements
59  basis_library_prefix (PREFIX ${LANGUAGE})
60  set (RE "[ \\t]*#[ \\t]*include[ \\t]+[<\"](${PREFIX})?basis.h[\">]") # e.g., #include "basis.h", #include <pkg/basis.h>
61  if (SCRIPT MATCHES "(^|\n)[ \t]*${RE}([ \t]*//.*|[ \t]*)(\n|$)")
62  set (UTILITIES_USED TRUE)
63  break ()
64  endif ()
65  # --------------------------------------------------------------------------
66  # Python/Jython
67  elseif (LANGUAGE MATCHES "[JP]YTHON")
68  # read script file
69  file (READ "${SOURCE_FILE}" SCRIPT)
70  # deprecated BASIS_PYTHON_UTILITIES macro
71  if (SCRIPT MATCHES "(^|\n|;)[ \t]*\@BASIS_PYTHON_UTILITIES\@")
72  message (FATAL_ERROR "Script ${SOURCE_FILE} uses the deprecated BASIS macro \@BASIS_PYTHON_UTILITIES\@!")
73  endif ()
74  basis_sanitize_for_regex (PYTHON_PACKAGE "${PROJECT_NAMESPACE_PYTHON}")
75  # match use of package-specific utilities
76  if (SCRIPT MATCHES "[^a-zA-Z._]${PYTHON_PACKAGE}.basis([.; \t\n]|$)") # e.g., basis = <package>.basis, <package>.basis.exedir()
77  set (UTILITIES_USED TRUE)
78  else ()
79  # match import statements
80  foreach (RE IN ITEMS
81  "import[ \\t]+basis" # e.g., import basis
82  "import[ \\t]+${PYTHON_PACKAGE}\\.basis" # e.g., import <package>.basis
83  "import[ \\t]+\\.\\.?basis" # e.g., import .basis, import ..basis
84  "from[ \\t]+${PYTHON_PACKAGE}[ \\t]+import[ \\t]+basis" # e.g., from <package> import basis
85  "from[ \\t]+${PYTHON_PACKAGE}.basis[ \\t]+import[ \\t].*" # e.g., from <package>.basis import which
86  "from[ \\t]+\\.\\.?[ \\t]+import[ \\t]+basis" # e.g., from . import basis", "from .. import basis
87  "from[ \\t]+\\.\\.?basis[ \\t]+import[ \\t].*" # e.g., from .basis import which, WhichError, from ..basis import which
88  ) # foreach RE
89  if (SCRIPT MATCHES "(^|\n|;)[ \t]*${RE}([ \t]*as[ \t]+.*)?([ \t]*#.*|[ \t]*)(;|\n|$)")
90  set (UTILITIES_USED TRUE)
91  break ()
92  endif ()
93  endforeach ()
94  endif ()
95  # --------------------------------------------------------------------------
96  # Perl
97  elseif (LANGUAGE MATCHES "PERL")
98  # read script file
99  file (READ "${SOURCE_FILE}" SCRIPT)
100  # deprecated BASIS_PERL_UTILITIES macro
101  if (SCRIPT MATCHES "(^|\n|;)[ \t]*\@BASIS_PERL_UTILITIES\@")
102  message (FATAL_ERROR "Script ${SOURCE_FILE} uses the deprecated BASIS macro \@BASIS_PERL_UTILITIES\@!")
103  endif ()
104  # match use/require statements
105  basis_sanitize_for_regex (PERL_PACKAGE "${PROJECT_NAMESPACE_PERL}")
106  set (RE "(use|require)[ \\t]+${PERL_PACKAGE}::Basis([ \\t]+.*)?") # e.g., use <Package>::Basis qw(:everything);
107  if (SCRIPT MATCHES "(^|\n|;)[ \t]*${RE}([ \t]*#.*|[ \t]*)(;|\n|$)")
108  set (UTILITIES_USED TRUE)
109  break ()
110  endif ()
111  # --------------------------------------------------------------------------
112  # Bash
113  elseif (LANGUAGE MATCHES "BASH")
114  # read script file
115  file (READ "${SOURCE_FILE}" SCRIPT)
116  # deprecated BASIS_BASH_UTILITIES macro
117  if (SCRIPT MATCHES "(^|\n|;)[ \t]*\@BASIS_BASH_UTILITIES\@")
118  message (FATAL_ERROR "Script ${SOURCE_FILE} uses the deprecated BASIS macro \@BASIS_BASH_UTILITIES\@!")
119  endif ()
120  # match source/. built-ins
121  set (RE "(source|\\.)[ \\t]+\\\"?\\\${?BASIS_BASH_UTILITIES}?\\\"?[ \\t]*(\\|\\|.*|&&.*)?(#.*)?") # e.g., . ${BASIS_BASH_UTILITIES} || exit 1
122  if (SCRIPT MATCHES "(^|\n|;)[ \t]*(${RE})[ \t]*(;|\n|$)")
123  set (UTILITIES_USED TRUE)
124  endif ()
125  endif ()
126  # return
127  set (${VAR} "${UTILITIES_USED}" PARENT_SCOPE)
128 endfunction ()

134 # ----------------------------------------------------------------------------
135 ## @brief Add build target for BASIS C++ utilities library.
136 #
137 # This function is called by basis_add_executable_target() and basis_add_library_target()
138 # in order to add the "basis" build target for the static project-specific
139 # BASIS utilities library for C++. If the target was added before, it is
140 # only used to get the target UID of this build target so the newly added
141 # executable or library can be linked to it.
142 #
143 # The CMake function add_library() checks if the specified source code files
144 # exist. If a source file is not found, an error is raised by CMake. The BASIS
145 # utilities can, however, only be configured at the end of the configuration
146 # step. Therefore, this function simply writes dummy C++ source files in order
147 # to pass the existence check. The actual source files are configured by the
148 # function basis_configure_utilities() which is called by basis_project_end().
149 #
150 # After writing these dummy source files, a library build target for the
151 # project-specific BASIS C++ utilities is added. This build target is not
152 # being build as part of the ALL target in case it is never used by any of
153 # the build targets of the project. Only if build target links to this
154 # library, it will be build and installed.
155 #
156 # @param [out] UID UID of added build target.
157 function (basis_add_cxx_utilities_library UID)
158  # target UID of "basis" library target
159  _basis_make_target_uid (TARGET_UID basis)
160  if (NOT TARGET ${TARGET_UID})
161  if (PROJECT_IS_SUBPROJECT)
162  # a subproject has it's own version of the project-specific BASIS utilities
163  # as the targets and functions live in a separate namespace
164  set (CODE_DIR "${BINARY_CODE_DIR}")
165  set (INCLUDE_DIR "${BINARY_INCLUDE_DIR}")
166  set (OUTPUT_DIR "${BINARY_ARCHIVE_DIR}")
167  set (INSTALL_DIR "${INSTALL_ARCHIVE_DIR}")
168  else ()
169  # modules, on the other side, share the library with the top-level project
170  # the addition of the utilities target is in this case only required because
171  # of the install(TARGETS) and install(EXPORT) commands.
172  set (CODE_DIR "${TOPLEVEL_BINARY_CODE_DIR}")
173  set (INCLUDE_DIR "${TOPLEVEL_BINARY_INCLUDE_DIR}")
174  set (OUTPUT_DIR "${TOPLEVEL_BINARY_ARCHIVE_DIR}")
175  set (INSTALL_DIR "${TOPLEVEL_INSTALL_ARCHIVE_DIR}")
176  endif ()
177  # write dummy source files
178  basis_library_prefix (PREFIX CXX)
179  foreach (S IN ITEMS basis.h basis.cxx)
180  if (S MATCHES "\\.h$")
181  set (S "${INCLUDE_DIR}/${PREFIX}${S}")
182  else ()
183  set (S "${CODE_DIR}/${S}")
184  endif ()
185  if (NOT EXISTS "${S}")
186  file (WRITE "${S}"
187  "#error This dummy source file should have been replaced by the"
188  " BASIS CMake function basis_configure_utilities()"
189  )
190  endif ()
191  endforeach ()
192  # add library target if not present yet - only build if required
193  add_library (${TARGET_UID} STATIC "${CODE_DIR}/basis.cxx")
194  # define dependency on non-project specific utilities as the order in
195  # which static libraries are listed on the command-line for the linker
196  # matters; this will help CMake to get the order right
197  target_link_libraries (${TARGET_UID} ${BASIS_CXX_UTILITIES_LIBRARY})
198  # set target properties
199  _set_target_properties (
200  ${TARGET_UID}
201  PROPERTIES
202  BASIS_TYPE STATIC_LIBRARY
203  OUTPUT_NAME basis
204  ARCHIVE_OUTPUT_DIRECTORY "${OUTPUT_DIR}"
205  ARCHIVE_INSTALL_DIRECTORY "${INSTALL_DIR}"
206  )
207  # export
208  basis_add_export_target (EXPORT_OPT ${TARGET_UID} FALSE ${INSTALL_DIR})
209  # installation
210  install (
211  TARGETS ${TARGET_UID} ${EXPORT_OPT}
212  ARCHIVE
213  DESTINATION "${INSTALL_DIR}"
214  COMPONENT "${BASIS_LIBRARY_COMPONENT}"
215  )
216  # debug message
217  if (BASIS_DEBUG)
218  message ("** Added BASIS utilities library ${TARGET_UID}")
219  endif ()
220  endif ()
221  # done
222  basis_set_project_property (PROPERTY PROJECT_USES_CXX_UTILITIES TRUE)
223  set (${UID} "${TARGET_UID}" PARENT_SCOPE)
224 endfunction ()

337 # ----------------------------------------------------------------------------
338 ## @brief Add build target for BASIS utilities library.
339 #
340 # This function is called by basis_target_link_libraries() in order to add the
341 # "basis" build target for the project-specific BASIS utilities for a given
342 # source code language. If the target was added before, it is only used to get
343 # the target UID of this build target so the executable or library can be
344 # linked to it.
345 #
346 # @param [out] UID UID of added build target.
347 # @param [in] LANGUAGE Programming language of utilities.
348 macro (basis_add_utilities_library UID LANGUAGE)
349  if ("^${LANGUAGE}$" STREQUAL "^CXX$")
350  if (NOT TARGET ${BASIS_CXX_UTILITIES_LIBRARY})
351  message (FATAL_ERROR "This project makes use of the BASIS C++ utilities"
352  " but BASIS was built without these utilities enabled.")
353  endif ()
354  basis_add_cxx_utilities_library (${UID})
355  elseif ("^${LANGUAGE}$" STREQUAL "^PYTHON$")
356  if (NOT TARGET ${BASIS_PYTHON_UTILITIES_LIBRARY})
357  message (FATAL_ERROR "This project makes use of the BASIS Python utilities"
358  " but BASIS was built without these utilities enabled.")
359  endif ()
360  basis_add_python_utilities_library (${UID})
361  elseif ("^${LANGUAGE}$" STREQUAL "^PERL$")
362  if (NOT TARGET ${BASIS_PERL_UTILITIES_LIBRARY})
363  message (FATAL_ERROR "This project makes use of the BASIS Perl utilities"
364  " but BASIS was built without these utilities enabled.")
365  endif ()
366  basis_add_perl_utilities_library (${UID})
367  elseif ("^${LANGUAGE}$" STREQUAL "^BASH$")
368  if (NOT TARGET ${BASIS_BASH_UTILITIES_LIBRARY})
369  message (FATAL_ERROR "This project makes use of the BASIS Bash utilities"
370  " but BASIS was built without these utilities enabled.")
371  endif ()
372  basis_add_bash_utilities_library (${UID})
373  else ()
374  message (FATAL_ERROR "Unsupported language: ${LANGUAGE}")
375  endif ()
376 endmacro ()

392 # ----------------------------------------------------------------------------
393 ## @brief Configure BASIS utilities.
394 #
395 # This function configures the following source files which can be used
396 # within the source code of the project. If the BASIS utilities for a specific
397 # language are not used by any of the project's build targets, no target for
398 # the build of these utilities is added, unless the
399 # @c BUILD_BASIS_UTILITIES_FOR_<LANGUAGE> option is set to @c ON. A reason
400 # for forcing the build of the BASIS utilities is that the libraries should
401 # be used by other projects which may want to make use of the BASIS Utilities
402 # to get access to the project attributes.
403 #
404 # <table border="0">
405 # <tr>
406 # @tp @b basis.h @endtp
407 # <td>Header file declaring the BASIS utilities for C++.</td>
408 # </tr>
409 # <tr>
410 # @tp @b basis.cxx @endtp
411 # <td>Definitions of the constants and functions declared in basis.h.</td>
412 # </tr>
413 # <tr>
414 # @tp @b basis.py @endtp
415 # <td>Module defining the BASIS utilities for Python.</td>
416 # </tr>
417 # <tr>
418 # @tp @b Basis.pm @endtp
419 # <td>Module defining the BASIS utilities for Perl.</td>
420 # </tr>
421 # <tr>
422 # @tp @b basis.sh @endtp
423 # <td>Module defining the BASIS utilities for Bash.</td>
424 # </tr>
425 # </table>
426 #
427 # @note Dummy versions of the C++ source files have been written by the
428 # function basis_add_utilities_library() beforehand. This is
429 # necessary because CMake's add_executable() and add_library() commands
430 # raise an error if any of the specified source files does not exist.
431 function (basis_configure_utilities)
432  set (SKIP TRUE)
433  foreach (L IN ITEMS CXX PYTHON PERL BASH)
434  if (BUILD_BASIS_UTILITIES_FOR_${L})
435  set (${L} ON)
436  else ()
437  basis_get_project_property (${L} PROPERTY PROJECT_USES_${L}_UTILITIES)
438  endif ()
439  if (${L})
440  set (SKIP FALSE)
441  endif ()
442  endforeach ()
443  if (SKIP)
444  return ()
445  endif ()
446  message (STATUS "Configuring BASIS utilities...")
447  # --------------------------------------------------------------------------
448  # executable target information
449  _basis_generate_executable_target_info (${CXX} ${PYTHON} ${PERL} ${BASH})
450  # --------------------------------------------------------------------------
451  # project ID -- used by print_version() in particular
452  set (PROJECT_ID "${PROJECT_PACKAGE_NAME}")
453  if (NOT PROJECT_NAME MATCHES "${PROJECT_PACKAGE_NAME}")
454  set (PROJECT_ID "${PROJECT_ID}, ${PROJECT_NAME}")
455  endif ()
456  # --------------------------------------------------------------------------
457  # C++
458  if (CXX)
459  # make sure that library target is added which is not the case yet
460  # if the BASIS C++ utilities are not used by any project target, but
461  # their build is forced via the BUILD_BASIS_UTILITIES_FOR_CXX option
462  basis_add_cxx_utilities_library (TARGET_UID)
463  # paths - build tree
464  set (BUILD_ROOT_PATH_CONFIG "${CMAKE_BINARY_DIR}")
465  set (RUNTIME_BUILD_PATH_CONFIG "${BINARY_RUNTIME_DIR}")
466  set (LIBEXEC_BUILD_PATH_CONFIG "${BINARY_LIBEXEC_DIR}")
467  set (LIBRARY_BUILD_PATH_CONFIG "${BINARY_LIBRARY_DIR}")
468  set (DATA_BUILD_PATH_CONFIG "${PROJECT_DATA_DIR}")
469  # paths - installation
470  file (RELATIVE_PATH RUNTIME_PATH_PREFIX_CONFIG "${CMAKE_INSTALL_PREFIX}/${INSTALL_RUNTIME_DIR}" "${CMAKE_INSTALL_PREFIX}")
471  string (REGEX REPLACE "/$|\\$" "" RUNTIME_PATH_PREFIX_CONFIG "${RUNTIME_PATH_PREFIX_CONFIG}")
472  file (RELATIVE_PATH LIBEXEC_PATH_PREFIX_CONFIG "${CMAKE_INSTALL_PREFIX}/${INSTALL_LIBEXEC_DIR}" "${CMAKE_INSTALL_PREFIX}")
473  string (REGEX REPLACE "/$|\\$" "" LIBEXEC_PATH_PREFIX_CONFIG "${LIBEXEC_PATH_PREFIX_CONFIG}")
474  set (RUNTIME_PATH_CONFIG "${INSTALL_RUNTIME_DIR}")
475  set (LIBEXEC_PATH_CONFIG "${INSTALL_LIBEXEC_DIR}")
476  set (LIBRARY_PATH_CONFIG "${INSTALL_LIBRARY_DIR}")
477  set (DATA_PATH_CONFIG "${INSTALL_DATA_DIR}")
478  # namespace
479  set (PROJECT_NAMESPACE_CXX_BEGIN "namespace ${PROJECT_PACKAGE_NAME_L} {")
480  set (PROJECT_NAMESPACE_CXX_END "}")
481  if (PROJECT_IS_SUBPROJECT)
482  set (PROJECT_NAMESPACE_CXX_BEGIN "${PROJECT_NAMESPACE_CXX_BEGIN} namespace ${PROJECT_NAME_L} {")
483  set (PROJECT_NAMESPACE_CXX_END "${PROJECT_NAMESPACE_CXX_END} }")
484  endif ()
485  # executable target information
486  set (EXECUTABLE_TARGET_INFO "${EXECUTABLE_TARGET_INFO_CXX}")
487  # configure source files
488  basis_library_prefix (PREFIX CXX)
489  configure_file ("${BASIS_CXX_TEMPLATES_DIR}/basis.h.in" "${BINARY_INCLUDE_DIR}/${PREFIX}basis.h" @ONLY)
490  configure_file ("${BASIS_CXX_TEMPLATES_DIR}/basis.cxx.in" "${BINARY_CODE_DIR}/basis.cxx" @ONLY)
491  source_group (BASIS FILES "${BINARY_INCLUDE_DIR}/${PREFIX}basis.h" "${BINARY_CODE_DIR}/basis.cxx")
492  endif ()
493  # --------------------------------------------------------------------------
494  # Python
495  if (PYTHON)
496  # utilities available?
497  if (NOT BASIS_UTILITIES_ENABLED MATCHES "PYTHON")
498  message (FATAL_ERROR "BASIS Python utilities required by this package"
499  " but BASIS was built without Python utilities."
500  " Rebuild BASIS with Python utilities enabled.")
501  endif ()
502  # make sure that library target is added which is not the case yet
503  # if the BASIS Python utilities are not used by any project target, but
504  # their build is forced via the BUILD_BASIS_UTILITIES_FOR_PYTHON option
505  basis_add_python_utilities_library (TARGET_UID)
506  # set target properties
507  set (SCRIPT_DEFINITIONS
508  "if (BUILD_INSTALL_SCRIPT)
509  set (EXECUTABLE_TARGET_INFO \"${EXECUTABLE_TARGET_INFO_PYTHON_I}\")
510  else ()
511  set (EXECUTABLE_TARGET_INFO \"${EXECUTABLE_TARGET_INFO_PYTHON_B}\")
512  endif ()"
513  )
514  if ("^${PROJECT_NAME}" STREQUAL "^BASIS")

612 # ----------------------------------------------------------------------------
613 ## @brief Generate code for initialization of executable target information.
614 #
615 # This macro generates the initialization code of the executable target
616 # information dictionaries for different supported programming languages.
617 # In case of C++, the source file has been configured and copied to the binary
618 # tree in a first configuration pass such that it could be used in basis_add_*()
619 # commands which check the existence of the arguments immediately.
620 # As the generation of the initialization code requires a complete list of
621 # build targets (cached in @c BASIS_TARGETS), this function has to be called
622 # after all targets have been added and finalized (in case of custom targets).
623 #
624 # @param [in] CXX Request code for C++.
625 # @param [in] PYTHON Request code for Python.
626 # @param [in] PERL Request code for Perl.
627 # @param [in] BASH Request code for Bash.
628 #
629 # @returns Sets the following variables for each requested language.
630 #
631 # @retval EXECUTABLE_TARGET_INFO_CXX C++ code for both build tree and installation.
632 # @retval EXECUTABLE_TARGET_INFO_PYTHON_B Python code for build tree.
633 # @retval EXECUTABLE_TARGET_INFO_PYTHON_I Python code for installation.
634 # @retval EXECUTABLE_TARGET_INFO_PERL_B Perl code for build tree.
635 # @retval EXECUTABLE_TARGET_INFO_PERL_I Perl code for installation.
636 # @retval EXECUTABLE_TARGET_INFO_BASH_B Bash code for build tree.
637 # @retval EXECUTABLE_TARGET_INFO_BASH_I Bash code for installation.
638 # @retval EXECUTABLE_TARGET_INFO_BASH_A Bash code to set aliases.
639 # @retval EXECUTABLE_TARGET_INFO_BASH_S Bash code to set short aliases.
640 function (_basis_generate_executable_target_info CXX PYTHON PERL BASH)
641  # --------------------------------------------------------------------------
642  if (NOT CXX AND NOT PYTHON AND NOT PERL AND NOT BASH)
643  return ()
644  endif ()
645  # --------------------------------------------------------------------------
646  # local constants
647  basis_sanitize_for_regex (PROJECT_NAMESPACE_CMAKE_RE "${PROJECT_NAMESPACE_CMAKE}")
648  # --------------------------------------------------------------------------
649  # lists of executable targets and their location
650  set (EXECUTABLE_TARGETS)
651  set (BUILD_LOCATIONS)
652  set (INSTALL_LOCATIONS)
653  # project targets
654  basis_get_project_property (TARGETS)
655  foreach (TARGET IN LISTS TARGETS)
656  basis_get_target_type (TYPE ${TARGET})
657  if (TYPE MATCHES "EXECUTABLE")
658  basis_get_target_location (BUILD_LOCATION ${TARGET} ABSOLUTE)
659  basis_get_target_location (INSTALL_LOCATION ${TARGET} POST_INSTALL)
660  if (BUILD_LOCATION)
661  list (APPEND EXECUTABLE_TARGETS "${TARGET}")
662  list (APPEND BUILD_LOCATIONS "${BUILD_LOCATION}")
663  list (APPEND INSTALL_LOCATIONS "${INSTALL_LOCATION}")
664  else ()
665  message (FATAL_ERROR "Failed to determine build location of target ${TARGET}!")
666  endif ()
667  endif ()
668  endforeach ()
669  # imported targets
670  basis_get_project_property (IMPORTED_TARGETS)
671  basis_get_project_property (IMPORTED_TYPES)
672  basis_get_project_property (IMPORTED_LOCATIONS)
673  set (I 0)
674  list (LENGTH IMPORTED_TARGETS N)
675  while (I LESS N)
676  list (GET IMPORTED_TARGETS ${I} TARGET)
677  list (GET IMPORTED_TYPES ${I} TYPE)
678  list (GET IMPORTED_LOCATIONS ${I} LOCATION)
679  if (TYPE MATCHES "EXECUTABLE")
680  # get corresponding UID (target may be imported from other module)
681  basis_get_target_uid (UID ${TARGET})
682  # skip already considered executables
683  list (FIND EXECUTABLE_TARGETS ${UID} IDX)
684  if (IDX EQUAL -1)
685  if (LOCATION MATCHES "^NOTFOUND$")
686  message (WARNING "Imported target ${TARGET} has no location property!")
687  else ()
688  list (APPEND EXECUTABLE_TARGETS ${TARGET})
689  list (APPEND BUILD_LOCATIONS "${LOCATION}")
690  list (APPEND INSTALL_LOCATIONS "${LOCATION}")
691  endif ()
692  endif ()
693  endif ()
694  math (EXPR I "${I} + 1")
695  endwhile ()
696  # --------------------------------------------------------------------------
697  # determine maximum length of target alias for prettier output
698  set (MAX_ALIAS_LENGTH 0)
699  foreach (TARGET_UID IN LISTS EXECUTABLE_TARGETS)
700  if (TARGET_UID MATCHES "^\\.")
701  string (LENGTH "${TARGET_UID}" LENGTH)
702  else ()
703  string (LENGTH "${TOPLEVEL_PROJECT_NAMESPACE_CMAKE}.${TARGET_UID}" LENGTH)
704  endif ()
705  if (LENGTH GREATER MAX_ALIAS_LENGTH)
706  set (MAX_ALIAS_LENGTH ${LENGTH})
707  endif ()
708  endforeach ()
709  # --------------------------------------------------------------------------
710  # generate source code
711  set (CC) # C++ - build tree and install tree version, constructor block
712  set (PY_B) # Python - build tree version
713  set (PY_I) # Python - install tree version
714  set (PL_B) # Perl - build tree version, hash entries
715  set (PL_I) # Perl - install tree version, hash entries
716  set (SH_B) # Bash - build tree version
717  set (SH_I) # Bash - install tree version
718  set (SH_A) # Bash - aliases
719  set (SH_S) # Bash - short aliases
720 
721  if (CXX)
722  set (CC "// the following code was automatically generated by the BASIS")
723  set (CC "${CC}\n // CMake function basis_configure_ExecutableTargetInfo()")
724  endif ()
725 
726  set (I 0)
727  list (LENGTH EXECUTABLE_TARGETS N)
728  while (I LESS N)
729  # ------------------------------------------------------------------------
730  # get executable information
731  list (GET EXECUTABLE_TARGETS ${I} TARGET_UID)
732  list (GET BUILD_LOCATIONS ${I} BUILD_LOCATION)
733  list (GET INSTALL_LOCATIONS ${I} INSTALL_LOCATION)
734  get_target_property (BUNDLED ${TARGET_UID} BUNDLED)
735  get_target_property (IMPORTED ${TARGET_UID} IMPORTED)
736  # insert $(IntDir) for Visual Studio build location
737  if (CMAKE_GENERATOR MATCHES "Visual Studio")
738  basis_get_target_type (TYPE ${TARGET_UID})
739  if (TYPE MATCHES "^EXECUTABLE$")
740  get_filename_component (DIRECTORY "${BUILD_LOCATION}" PATH)
741  get_filename_component (FILENAME "${BUILD_LOCATION}" NAME)
742  set (BUILD_LOCATION_WITH_INTDIR "${DIRECTORY}/$(IntDir)/${FILENAME}")
743  else ()
744  set (BUILD_LOCATION_WITH_INTDIR "${BUILD_LOCATION}")
745  endif ()
746  else ()
747  set (BUILD_LOCATION_WITH_INTDIR "${BUILD_LOCATION}")
748  endif ()
749  # installation path (relative) to different library paths
750  foreach (L LIBRARY PYTHON_LIBRARY PERL_LIBRARY BASH_LIBRARY)
751  if (INSTALL_LOCATION AND (NOT IMPORTED OR BUNDLED))
752  file (
753  RELATIVE_PATH INSTALL_LOCATION_REL2${L}
754  "${CMAKE_INSTALL_PREFIX}/${INSTALL_${L}_DIR}/<package>"
755  "${INSTALL_LOCATION}"
756  )
757  else ()
758  set (INSTALL_LOCATION_REL2${L} "${INSTALL_LOCATION}")
759  endif ()
760  endforeach ()
761  # target UID including project namespace
762  basis_get_fully_qualified_target_uid (ALIAS "${TARGET_UID}")
763  # indentation after dictionary key, i.e., alias
764  string (LENGTH "${ALIAS}" ALIAS_LENGTH)
765  math (EXPR NUM "${MAX_ALIAS_LENGTH} - ${ALIAS_LENGTH} + 1")
766  if (NUM GREATER 1)
767  string (RANDOM LENGTH ${NUM} ALPHABET " " S)
768  else ()
769  set (S " ")
770  endif ()
771  # ------------------------------------------------------------------------
772  # C++
773  if (CXX)
774  get_filename_component (EXEC_NAME "${BUILD_LOCATION}" NAME)
775  get_filename_component (BUILD_DIR "${BUILD_LOCATION}" PATH)
776  if (INSTALL_LOCATION)
777  get_filename_component (INSTALL_DIR "${INSTALL_LOCATION}" PATH)
778  endif ()
779 
780  set (CC "${CC}\n")
781  set (CC "${CC}\n // ${TARGET_UID}")
782  set (CC "${CC}\n _exec_names [\"${ALIAS}\"]${S}= \"${EXEC_NAME}\";")
783  set (CC "${CC}\n _build_dirs [\"${ALIAS}\"]${S}= \"${BUILD_DIR}\";")
784  set (CC "${CC}\n _install_dirs[\"${ALIAS}\"]${S}= \"${INSTALL_DIR}\";")
785  endif ()
786  # ------------------------------------------------------------------------
787  # Python
788  if (PYTHON)
789  set (PY_B "${PY_B} '${ALIAS}':${S}'${BUILD_LOCATION_WITH_INTDIR}',\n")
790  if (INSTALL_LOCATION)
791  set (PY_I "${PY_I} '${ALIAS}':${S}'${INSTALL_LOCATION_REL2PYTHON_LIBRARY}',\n")
792  else ()
793  set (PY_I "${PY_I} '${ALIAS}':${S}'',\n")
794  endif ()
795  endif ()
796  # ------------------------------------------------------------------------
797  # Perl
798  if (PERL)
799  if (PL_B)
800  set (PL_B "${PL_B},\n")
801  endif ()
802  set (PL_B "${PL_B} '${ALIAS}'${S}=> '${BUILD_LOCATION_WITH_INTDIR}'")
803  if (PL_I)
804  set (PL_I "${PL_I},\n")
805  endif ()
806  if (INSTALL_LOCATION)
807  set (PL_I "${PL_I} '${ALIAS}'${S}=> '${INSTALL_LOCATION_REL2PERL_LIBRARY}'")
808  else ()
809  set (PL_I "${PL_I} '${ALIAS}'${S}=> ''")
810  endif ()
811  endif ()
812  # ------------------------------------------------------------------------
813  # Bash
814  if (BASH)
815  # hash entry
816  set (SH_B "${SH_B}\n _basis_executabletargetinfo_add '${ALIAS}'${S}LOCATION '${BUILD_LOCATION}'")
817  if (INSTALL_LOCATION)
818  set (SH_I "${SH_I}\n _basis_executabletargetinfo_add '${ALIAS}'${S}LOCATION '${INSTALL_LOCATION_REL2BASH_LIBRARY}'")
819  else ()
820  set (SH_I "${SH_I}\n _basis_executabletargetinfo_add '${ALIAS}'${S}LOCATION ''")
821  endif ()
822  # alias
823  set (SH_A "${SH_A}\n alias '${ALIAS}'=`get_executable_path '${ALIAS}'`")
824  # short alias (if target belongs to this project)
825  if (ALIAS MATCHES "^${PROJECT_NAMESPACE_CMAKE_RE}\\.")
826  basis_get_target_name (TARGET_NAME "${ALIAS}")
827  set (SH_S "${SH_S}\n alias '${TARGET_NAME}'='${ALIAS}'")
828  endif ()
829  endif ()
830  # ------------------------------------------------------------------------
831  # next executable target
832  math (EXPR I "${I} + 1")
833  endwhile ()
834  # --------------------------------------------------------------------------
835  # remove unnecessary leading newlines
836  string (STRIP "${CC}" CC)
837  string (STRIP "${PY_B}" PY_B)
838  string (STRIP "${PY_I}" PY_I)
839  string (STRIP "${PL_B}" PL_B)
840  string (STRIP "${PL_I}" PL_I)
841  string (STRIP "${SH_B}" SH_B)
842  string (STRIP "${SH_I}" SH_I)
843  string (STRIP "${SH_A}" SH_A)
844  string (STRIP "${SH_S}" SH_S)
845  # --------------------------------------------------------------------------
846  # return
847  set (EXECUTABLE_TARGET_INFO_CXX "${CC}" PARENT_SCOPE)
848  set (EXECUTABLE_TARGET_INFO_PYTHON_B "${PY_B}" PARENT_SCOPE)
849  set (EXECUTABLE_TARGET_INFO_PYTHON_I "${PY_I}" PARENT_SCOPE)
850  set (EXECUTABLE_TARGET_INFO_PERL_B "${PL_B}" PARENT_SCOPE)
851  set (EXECUTABLE_TARGET_INFO_PERL_I "${PL_I}" PARENT_SCOPE)
852  set (EXECUTABLE_TARGET_INFO_BASH_B "${SH_B}" PARENT_SCOPE)
853  set (EXECUTABLE_TARGET_INFO_BASH_I "${SH_I}" PARENT_SCOPE)
854  set (EXECUTABLE_TARGET_INFO_BASH_A "${SH_A}" PARENT_SCOPE)
855  set (EXECUTABLE_TARGET_INFO_BASH_S "${SH_S}" PARENT_SCOPE)
856 endfunction ()
