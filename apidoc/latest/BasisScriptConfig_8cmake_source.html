11 # @file BasisScriptConfig.cmake
12 # @brief BASIS script configuration.
13 #
14 # @note The file BasisScriptConfig.cmake is automatically generated by BASIS
15 # from the template file ScriptConfig.cmake.in which is part of BASIS.
16 #
17 # This script configuration file is included before the default script
18 # configuration file which is part of the package itself (ScriptConfig.cmake),
19 # if such file exists in the @c PROJECT_CONFIG_DIR.
20 # Therefore, the BASIS script configuration can be overwritten by the
21 # default script configuration of the project, which in turn can be
22 # overwritten specifically for each script target by specifying a
23 # script target-specific script configuration using the @c CONFIG and/or
24 # @c CONFIG_FILE option of the basis_add_executable() or basis_add_library()
25 # command, respectively.
26 #
27 # @sa http://opensource.andreasschuh.com/cmake-basis/standard/scripttargets.html#script-configuration
