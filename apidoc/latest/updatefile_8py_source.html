7 # @file updatefile.py
8 # @brief Update file from template file while preserving custom sections (deprecated).
9 #
10 # This script is used by the BasisUpdate.cmake module. This module is used to
11 # update files of a project instantiated from a particular revision of the
12 # BASIS project template during the configure step of CMake. This way,
13 # projects pull the changes of the compatible template automatically.
14 # Sections in the original file which are enclosed by the tokens
15 # <basis-custom> and </basis-custom> or <basis-license> and
16 # </basis-license> (without trailing spaces) are preserved while all other
17 # content is replaced by the template file. The customized sections are
18 # inserted into the template in the order they appear in the original file
19 # and the template file. If more custom sections are present in the original
20 # file than in the template file, these custom sections are appended at the
21 # end of the resulting file.
22 #
23 # See the documentation of BasisUpdate.cmake for further details.

85 def extractCustomizedSections (input):
86  custom = []
87  start = 0
88  while True:
89  start = input.find (tokenCustomStart, start)
90  if start == -1:
91  break
92  start += len (tokenCustomStart)
93  next = input.find (tokenCustomStart, start)
94  end = input.find (tokenCustomEnd, start)
95  if end == -1 or (next != -1 and end > next):
96  print "WARNING: Found begin of customized section without end token '" + tokenCustomEnd + "'"
97  print "WARNING: Will keep template section instead"
98  custom.append (tokenKeepTemplate)
99  else:
100  custom.append (input [start:end])
101  start = next
102  return custom

105 def replaceCustomizedSections (input, custom):
106  start = 0
107  end = -1

125  end = start + len (section)

127  start = end + len (tokenCustomEnd)

191  sections = extractLicenseSections (input)

195  # check if content changed
