48  if ($type =~ s/^seq_//) {
49  if ($type eq 'text') {
50  return "$item";
51  }
52  if ($type = $MARKUP->{ $type }) {
53  if ($item =~ /[<>]/) {

59  }
60  }
61  elsif (ref $item eq 'HASH') {
62  if (defined $item->{ content }) {
63  return $item->{ content }->present($self);
64  }
65  elsif (defined $item->{ text }) {
66  my $text = $item->{ text };
67  return ref $text ? $text->present($self) : $text;
68  }
69  else {
70  return '';
71  }
72  }
73  elsif (! ref $item) {
74  return $item;
