31 # The statement "from basis import utilities" (or "import basis.utilities")
32 # cannot be used here with Python 2.x because the name of this module is
33 # the same as the name of the basis package. Therefore, load the source
34 # file of the basis.utilities module explicitly here.
35 #
36 # Note: imp.load_source() would be quite more compact. However, the following
37 # will also still work with Python 3.1 and 3.2. For Python 3.3 a
38 # different implementation might be necessary using loader objects.
39 import imp
40 # path to BASIS Utilities modules
41 _path = os.getenv('BASIS_PYTHONPATH')
42 if _path is None:
43  for _syspath in sys.path:
44  if (os.path.isfile(os.path.join(_syspath, 'basis/utilities.py' )) or
45  os.path.isfile(os.path.join(_syspath, 'basis/utilities.pyc'))):
46  _path = _syspath
47  break
48 if _path is None:
49  _path = '..'
50  if not os.path.isabs(_path):
51  _path = os.path.realpath(os.path.join(os.path.dirname(__file__), _path))
52 if (not os.path.isfile(os.path.join(_path, 'basis/utilities.py' )) and
53  not os.path.isfile(os.path.join(_path, 'basis/utilities.pyc'))):
54  raise "Module basis.utilities not found at " + _path + "!\nSpecify path using the PYTHONPATH or BASIS_PYTHONPATH environment variable."
55 # basis/__init__.py must be loaded first
56 (_file, _filepath, _desc) = imp.find_module('basis', [_path])
57 try:
58  basis = imp.load_module('basis', _file, _filepath, _desc)
59 finally:
60  if _file: _file.close()
61 # then we can load basis.utilities
62 (_file, _filepath, _desc) = imp.find_module('utilities', [os.path.join(_path, 'basis')])
63 try:
64  utilities = imp.load_module('basis.utilities', _file, _filepath, _desc)
65 finally:
66  if _file: _file.close()
67 # clean up scope
68 del _path
69 del _syspath
70 del _file
71 del _filepath
72 del _desc
