67 # ----------------------------------------------------------------------------
68 # private helper macro to look for a particular required or optional AFNI program
69 macro (_AFNI_find_program NAME REQUIRED)
70  if (AFNI_DIR)
71  find_program (AFNI_${NAME}_EXECUTABLE NAMES ${NAME} HINTS ${AFNI_DIR} PATH_SUFFIXES bin NO_DEFAULT_PATH)
72  else ()
73  find_program (AFNI_${NAME}_EXECUTABLE NAMES ${NAME})
74  endif ()
75  if (NOT AFNI_${NAME}_EXECUTABLE)
76  if (AFNI_FIND_COMPONENTS)
77  # looking either only for required components or for both optional and
78  # and required components; in this case, let AFNI_FOUND reflect only
79  # whether all required components were found, but ignore the optional ones;
80  # the caller can still check AFNI_<tool>_EXECUTABLE explicitly for these
81  # optional components to see whether or not a particular AFNI programs was found
82  if (REQUIRED)
83  set (AFNI_FOUND FALSE)
84  endif ()
85  else ()
86  # looking only for optional components anyway, so let AFNI_FOUND reflect
87  # if all of these optional components were found instead
88  set (AFNI_FOUND FALSE)
89  endif ()
90  if (REQUIRED)
91  list (APPEND _AFNI_MISSING_COMPONENTS ${NAME})
92  else ()
93  list (APPEND _AFNI_MISSING_OPTIONAL_COMPONENTS ${NAME})
94  endif ()
95  endif ()
96  mark_as_advanced (AFNI_${NAME}_EXECUTABLE)
97 endmacro ()
