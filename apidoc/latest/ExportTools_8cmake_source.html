29 ## @brief Add target to export set.
30 #
31 # Targets of a subproject are added to the export set named after the subproject
32 # itself. All other targets, including in particular those of other modules,
33 # are added to the export set named after the top-level project.
34 #
35 # @param[out] EXPORT_OPTION Export option for install() command including
36 # the EXPORT option name. Set to an empty string
37 # if target is not installed.
38 # @param[in] TEST Whether given target is a test executable or library.
39 # @param[in] ARGN Optional installation destinations.
40 function (basis_add_export_target EXPORT_OPTION TARGET_UID TEST)
41  if (PROJECT_IS_SUBPROJECT)
42  set (EXPORT_SET "${PROJECT_NAME}")
43  else ()
44  set (EXPORT_SET "${TOPLEVEL_PROJECT_NAME}")
45  endif ()
46  if (TEST)
47  basis_set_project_property (PROJECT "${EXPORT_SET}" APPEND PROPERTY TEST_EXPORT_TARGETS "${TARGET_UID}")
48  else ()
49  basis_set_project_property (PROJECT "${EXPORT_SET}" APPEND PROPERTY EXPORT_TARGETS "${TARGET_UID}")
50  if (ARGN)
51  basis_set_project_property (PROJECT "${EXPORT_SET}" APPEND PROPERTY INSTALL_EXPORT_TARGETS "${TARGET_UID}")
52  endif ()
53  set (EXPORT_OPT "EXPORT;${EXPORT_SET}" PARENT_SCOPE)
54  endif ()
55 endfunction ()

79 # ----------------------------------------------------------------------------
80 ## @brief Get soname of object file.
81 #
82 # This function extracts the soname from object files in the ELF format on
83 # systems where the objdump command is available. On all other systems,
84 # an empty string is returned.
85 #
86 # @param [out] SONAME The soname of the object file.
87 # @param [in] OBJFILE Object file in ELF format.
88 function (basis_get_soname SONAME OBJFILE)
89  # get absolute path of object file
90  basis_get_target_uid (TARGET_UID ${OBJFILE})
91  if (TARGET TARGET_UID)
92  basis_get_target_location (OBJFILE ${TARGET_UID} ABSOLUTE)
93  else ()
94  get_filename_component (OBJFILE "${OBJFILE}" ABSOLUTE)
95  endif ()
96  # usually CMake did this already
97  find_program (CMAKE_OBJDUMP NAMES objdump DOC "The objdump command")
98  # run objdump and extract soname
99  execute_process (
100  COMMAND ${CMAKE_OBJDUMP} -p "${OBJFILE}"
101  COMMAND sed -n "-e's/^[[:space:]]*SONAME[[:space:]]*//p'"
102  RESULT_VARIABLE STATUS
103  OUTPUT_VARIABLE SONAME_OUT
104  ERROR_QUIET
105  )
106  # return
107  if (STATUS EQUAL 0)
108  set (${SONAME} "${SONAME_OUT}" PARENT_SCOPE)
109  else ()
110  set (${SONAME} "" PARENT_SCOPE)
111  endif ()
112 endfunction ()

143 # ----------------------------------------------------------------------------
144 ## @brief Add code to add import targets.
145 function (basis_export_import_targets CODE)
146  set (C)
147  foreach (T IN LISTS ARGN)
148  basis_get_fully_qualified_target_uid (UID "${T}")
149  set (C "${C}\n# Create import target \"${UID}\"\n")
150  get_target_property (BASIS_TYPE ${T} "BASIS_TYPE")
151  if (BASIS_TYPE MATCHES "EXECUTABLE")
152  set (C "${C}add_executable (${UID} IMPORTED)\n")
153  elseif (BASIS_TYPE MATCHES "LIBRARY|MODULE|MEX")
154  if (BASIS_TYPE MATCHES "MEX|MCC")
155  set (TYPE SHARED)
156  elseif (BASIS_TYPE MATCHES "SCRIPT")
157  set (TYPE UNKNOWN)
158  else ()
159  string (REGEX REPLACE "_LIBRARY" "" TYPE "${BASIS_TYPE}")
160  endif ()
161  set (C "${C}add_library (${UID} ${TYPE} IMPORTED)\n")
162  else ()
163  message (FATAL_ERROR "Cannot export target ${T} of type ${BASIS_TYPE}! Use NOEXPORT option.")
164  endif ()
165  set (C "${C}set_target_properties (${UID} PROPERTIES BASIS_TYPE \"${BASIS_TYPE}\")\n")
166  endforeach ()
167  set (${CODE} "${${CODE}}${C}" PARENT_SCOPE)
168 endfunction ()

171 ## @brief Add code to set properties of imported targets for build tree.
172 function (basis_export_build_properties CODE)
173  set (C)
174  if (CMAKE_BUILD_TYPE)
175  set (CONFIG "${CMAKE_BUILD_TYPE}")
176  else ()
177  set (CONFIG "noconfig")
178  endif ()
179  string (TOUPPER "${CONFIG}" CONFIG_U)
180  foreach (T IN LISTS ARGN)
181  basis_get_fully_qualified_target_uid (UID "${T}")
182  get_target_property (BASIS_TYPE ${T} BASIS_TYPE)
183  set (C "${C}\n# Import target \"${UID}\" for configuration \"${CONFIG}\"\n")
184  set (C "${C}set_property (TARGET ${UID} APPEND PROPERTY IMPORTED_CONFIGURATIONS ${CONFIG_U})\n")
185  set (C "${C}set_target_properties (${UID} PROPERTIES\n")
186  if (BASIS_TYPE MATCHES "SCRIPT|LIBRARY|MEX")
187  get_target_property (LANGUAGE ${T} LANGUAGE)
188  if (LANGUAGE)
189  set (C "${C} IMPORTED_LINK_INTERFACE_LANGUAGES_${CONFIG_U} \"${LANGUAGE}\"\n")
190  endif ()
191  get_target_property (LINK_DEPENDS ${T} LINK_DEPENDS)
192  set (LINK_UIDS)
193  foreach (LINK_DEPEND IN LISTS LINK_DEPENDS)
194  basis_get_fully_qualified_target_uid (LINK_UID ${LINK_DEPEND})
195  list (APPEND LINK_UIDS "${LINK_UID}")
196  endforeach ()
197  set (C "${C} IMPORTED_LINK_INTERFACE_LIBRARIES_${CONFIG_U} \"${LINK_UIDS}\"\n")
198  endif ()
199  basis_get_target_location (LOCATION ${T} ABSOLUTE)
200  set (C "${C} IMPORTED_LOCATION_${CONFIG_U} \"${LOCATION}\"\n")
201  set (C "${C} )\n")
202  endforeach ()
203  set (${CODE} "${${CODE}}${C}" PARENT_SCOPE)
204 endfunction ()

253 # ----------------------------------------------------------------------------
254 ## @brief Export all targets added by basis_add_* commands.
255 function (basis_export_targets)
256  # parse arguments
257  CMAKE_PARSE_ARGUMENTS (ARGN "" "FILE;CUSTOM_FILE" "" ${ARGN})
258 
259  if (NOT ARGN_FILE)
260  message (FATAL_ERROR "basis_export_targets(): FILE option is required!")
261  endif ()
262  if (NOT ARGN_CUSTOM_FILE)
263  message (FATAL_ERROR "basis_export_targets(): CUSTOM_FILE option is required!")
264  endif ()
265 
266  if (IS_ABSOLUTE ${ARGN_FILE})
267  message (FATAL_ERROR "basis_export_targets(): FILE option argument must be a relative path!")
268  endif ()
269  if (IS_ABSOLUTE ${ARGN_CUSTOM_FILE})
270  message (FATAL_ERROR "basis_export_targets(): CUSTOM_FILE option argument must be a relative path!")
271  endif ()
272 
273  # --------------------------------------------------------------------------
274  # export non-custom targets
275  basis_get_project_property (EXPORT_TARGETS)
276  if (EXPORT_TARGETS)
277  if (BASIS_USE_TARGET_UIDS AND BASIS_USE_FULLY_QUALIFIED_UIDS)
278  set (NAMESPACE_OPT)
279  elseif (TOPLEVEL_PROJECT_NAMESPACE_CMAKE)
280  set (NAMESPACE_OPT NAMESPACE "${TOPLEVEL_PROJECT_NAMESPACE_CMAKE}.")
281  endif ()
282  export (
283  TARGETS ${EXPORT_TARGETS}
284  FILE "${PROJECT_BINARY_DIR}/${ARGN_FILE}"
285  ${NAMESPACE_OPT}
286  )
287  basis_get_project_property (INSTALL_EXPORT_TARGETS)
288  if (INSTALL_EXPORT_TARGETS)
289  foreach (COMPONENT "${BASIS_RUNTIME_COMPONENT}" "${BASIS_LIBRARY_COMPONENT}")
290  install (
291  EXPORT "${PROJECT_NAME}"
292  DESTINATION "${INSTALL_CONFIG_DIR}"
293  FILE "${ARGN_FILE}"
294  COMPONENT "${COMPONENT}"
295  ${NAMESPACE_OPT}
296  )
297  endforeach ()
298  endif ()
299  endif ()
300 
301  # --------------------------------------------------------------------------
302  # export custom targets and/or test targets
303  basis_get_project_property (CUSTOM_EXPORT_TARGETS)
304  basis_get_project_property (TEST_EXPORT_TARGETS)
305 
306  if (CUSTOM_EXPORT_TARGETS OR TEST_EXPORT_TARGETS)
307 
308  # write exports for build tree
309  basis_export_header (CONTENT)
310  basis_export_import_targets (CONTENT ${CUSTOM_EXPORT_TARGETS} ${TEST_EXPORT_TARGETS})
311  basis_export_build_properties (CONTENT ${CUSTOM_EXPORT_TARGETS} ${TEST_EXPORT_TARGETS})
312  basis_export_footer (CONTENT)
313 
314  file (WRITE "${PROJECT_BINARY_DIR}/${ARGN_CUSTOM_FILE}" "${CONTENT}")
315  unset (CONTENT)
316 
317  # write exports for installation - excluding test targets
318  if (CUSTOM_EXPORT_TARGETS)
319  set (INSTALL_EXPORT_FILE "${PROJECT_BINARY_DIR}/CMakeFiles/Export/${INSTALL_CONFIG_DIR}/${ARGN_CUSTOM_FILE}")
320 
321  basis_export_header (CONTENT)
322  basis_export_prefix (CONTENT)
323  basis_export_import_targets (CONTENT ${CUSTOM_EXPORT_TARGETS})
324  basis_export_install_properties (CONTENT ${CUSTOM_EXPORT_TARGETS})
325  basis_export_footer (CONTENT)
326 
327  file (WRITE "${INSTALL_EXPORT_FILE}" "${CONTENT}")
328  unset (CONTENT)
329 
330  install (
331  FILES "${INSTALL_EXPORT_FILE}"
332  DESTINATION "${INSTALL_CONFIG_DIR}"
333  )
334  endif ()
335 
336  endif ()
337 endfunction ()
