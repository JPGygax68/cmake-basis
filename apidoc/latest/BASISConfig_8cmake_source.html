1 ##############################################################################
2 # @file BASISConfig.cmake
3 # @brief CMake package configuration file.
4 #
5 # @note The BASISConfig.cmake file is automatically generated
6 # by BASIS from the template file Config.cmake.in which is part of
7 # the BASIS package.
8 #
9 # Variables in this file have to be prefixed by the name of this package.
10 # In the default package configuration file of BASIS, which can be included
11 # here using the \@BASIS_CONFIG\@ variable, this prefix is defined as @c NS
12 # variable. Moreover, the macro basis_set_config() should be used for
13 # convenience to set any configuration variable, e.g.,
14 #
15 # @code
16 # basis_set_config (INCLUDE_DIR "\@INCLUDE_DIR_CONFIG\@")
17 # @endcode
18 #
19 # where the variable @c INCLUDE_DIR_CONFIG is set in the ConfigSettings.cmake
20 # file of the BASIS installation. Note that this statement will set the
21 # variable ${NS}INCLUDE_DIR, where by default @c NS is the package name
22 # followed by an underscore (_).
23 #
24 # See the Config.cmake.in file of BASIS for details.
25 #

154 foreach (${NS}MODULE IN LISTS ${NS}MODULES_REQUESTED)
155  # check if requested module is available
156  list (FIND ${NS}MODULES_ENABLED ${NS}MODULE IDX)
157  if (IDX EQUAL -1)
158  message (FATAL_ERROR "Requested module ${${NS}MODULE} of package BASIS"
159  " was not enabled when building BASIS. Rebuild"
160  " the BASIS package with MODULE_${${NS}MODULE}"
161  " set to ON and try again.")
162  endif ()
163  # include configuration files of modules
164  set (${${NS}MODULE}_CONFIG_PREFIX "${NS}${MODULE}_")
165  include ("lib/cmake/basis/BASIS${${NS}MODULE}Config.cmake")
166  unset (${${NS}MODULE}_CONFIG_PREFIX)
167  # append include directories
168  if (${NS}${${NS}MODULE}_INCLUDE_DIRS)
169  basis_append_config (INCLUDE_DIRS ${${NS}${${NS}MODULE}_INCLUDE_DIRS})
170  elseif (${NS}${M}_INCLUDE_DIR)
171  basis_append_config (INCLUDE_DIRS ${${NS}${${NS}MODULE}_INCLUDE_DIR})
172  endif ()
173  # append library directories
174  if (${NS}${${NS}MODULE}_LIBRARY_DIRS)
175  basis_append_config (LIBRARY_DIRS ${${NS}${${NS}MODULE}_LIBRARY_DIRS})
176  elseif (${NS}${${NS}MODULE}_LIBARY_DIR)
177  basis_append_config (LIBRARY_DIRS ${${NS}${${NS}MODULE}_LIBRARY_DIR})
178  endif ()
179 endforeach ()
