92 ## @brief Find Python module.
93 #
94 # If the @c PYTHON_EXECUTABLE variable is set, this function at first tries
95 # to launch the Python interpreter, import the named Python module, and then
96 # determines the search path for the Python module from the @c __file__
97 # attribute of the loaded module. Otherwise, or if this fails, it looks for
98 # the Python module in the directories listed in the @c PYTHONPATH variable
99 # if defined. If this variable is not defined, the @c PYTHONPATH environment
100 # variable is used instead.
101 #
102 # @param [in] CACHEVAR Name of CMake cache variable which stores path of
103 # directory of the Python module. If not set or if
104 # the cache entry contains space characters only or
105 # ends in the string NOTFOUND, this function looks for
106 # the specified Python module. Otherwise, it does nothing
107 # and leaves the cache entry unmodified.
108 # @param [in] ARGN The remaining arguments are parsed and the following
109 # options extract:
110 # @par
111 # <table border=0>
112 # <tr>
113 # @tp @b NAME module @endtp
114 # <td>Name of the Python module.</td>
115 # </tr>
116 # <tr>
117 # @tp @b PYTHON_EXECUTABLE python @endtp
118 # <td>Full path of the Python interpreter executable. If not specified
119 # the global PYTHON_EXECUTABLE CMake variable/cache entry is used.</td>
120 # </tr>
121 # <tr>
122 # @tp @b PATH dir1 [dir2...] @endtp
123 # <td>Directories where to look for Python module.</td>
124 # </tr>
125 # <tr>
126 # @tp @b NO_PYTHONPATH @endtp
127 # <td>Do not consider the @c PYTHONPATH environment variable.</td>
128 # </tr>
129 # <tr>
130 # @tp @b NO_DEFAULT_PATH @endtp
131 # <td>Do not look in any default path such as the directories listed by the
132 # @c PYTHONPATH environment variable.</td>
133 # </tr>
134 # </table>
135 #
136 # @returns Sets the named cache variable of type @c PATH to the absolute path
137 # of the directory containing the specified Python @p MODULE if found,
138 # or the string "&lt;MODULE%gt;-NOTFOUND" otherwise.
139 function (basis_find_python_module CACHEVAR)
140  # do nothing if path of module already known from previous run
141  if (DEFINED ${CACHEVAR} AND NOT ${CACHEVAR} MATCHES "NOTFOUND$")
142  return ()
143  endif ()
144  # parse arguments
145  CMAKE_PARSE_ARGUMENTS (
146  ARGN
147  "NO_DEFAULT_PATH;NO_PYTHONPATH"
148  "PYTHON_EXECUTABLE;NAME"
149  "PATH"
150  ${ARGN}
151  )
152  if (NOT ARGN_NAME)
153  message ("basis_find_python_module(): Missing NAME argument!")
154  endif ()
155  if (ARGN_UNPARSED_ARGUMENTS)
156  message ("basis_find_python_module(): Invalid arguments: ${ARGN_UNPARSED_ARGUMENTS}")
157  endif ()
158  if (NOT ARGN_PYTHON_EXECUTABLE)
159  set (ARGN_PYTHON_EXECUTABLE "${PYTHON_EXECUTABLE}")
160  endif ()
161  if (ARGN_NO_DEFAULT_PATH)
162  set (ARGN_NO_PYTHONPATH TRUE)
163  set (ARGN_PYTHON_EXECUTABLE)
164  endif ()
165  # set initial value of cache entry
166  if (${CACHEVAR} MATCHES "^ *$")
167  set (${CACHEVAR} "${ARGN_NAME}-NOTFOUND" CACHE PATH "Directory containing ${ARGN_NAME} Python module/package." FORCE)
168  else ()
169  set (${CACHEVAR} "${ARGN_NAME}-NOTFOUND" CACHE PATH "Directory containing ${ARGN_NAME} Python module/package.")
170  endif ()
171  # 1. search specified paths
172  foreach (P ${ARGN_PATH}) # ignore empty entries
173  if (IS_ABSOLUTE "${P}")
174  if (EXISTS "${P}/${ARGN_NAME}.py" OR EXISTS "${P}/${ARGN_NAME}/__init__.py" OR
175  EXISTS "${P}/${ARGN_NAME}.pyc" OR EXISTS "${P}/${ARGN_NAME}/__init__.pyc")
176  set_property (CACHE ${CACHEVAR} PROPERTY VALUE "${P}")
177  return ()
178  endif ()
179  endif ()
180  endforeach ()
181  # 2. get __file__ attribute of module loaded in Python
182  if (ARGN_PYTHON_EXECUTABLE)
183  set (IMPORT_SITE_ERROR FALSE)
184  # 2a. try it with -E option -- the preferred way to run Python
185  execute_process (
186  COMMAND "${ARGN_PYTHON_EXECUTABLE}" -E -c "import ${ARGN_NAME}; print ${ARGN_NAME}.__file__"
187  RESULT_VARIABLE STATUS
188  OUTPUT_VARIABLE P
189  ERROR_VARIABLE ERROR
190  OUTPUT_STRIP_TRAILING_WHITESPACE
191  )
192  if (ERROR MATCHES "'import site' failed|ImportError: No module named site")
193  set (IMPORT_SITE_ERROR TRUE)
194  endif ()
195  # 2b. try it without -E option
196  if (NOT STATUS EQUAL 0)
197  execute_process (
198  COMMAND "${ARGN_PYTHON_EXECUTABLE}" -c "import ${ARGN_NAME}; print ${ARGN_NAME}.__file__"
199  RESULT_VARIABLE STATUS
200  OUTPUT_VARIABLE P
201  ERROR_VARIABLE ERROR
202  OUTPUT_STRIP_TRAILING_WHITESPACE
203  )
204  if (ERROR MATCHES "'import site' failed|ImportError: No module named site")
205  set (IMPORT_SITE_ERROR TRUE)
206  endif ()
207  if (NOT STATUS EQUAL 0 AND ERROR MATCHES "ImportError: No module named site")
208  set (PYTHONHOME "$ENV{PYTHONHOME}")
209  unset (ENV{PYTHONHOME})
210  execute_process (
211  COMMAND "${ARGN_PYTHON_EXECUTABLE}" -c "import ${ARGN_NAME}; print ${ARGN_NAME}.__file__"
212  RESULT_VARIABLE STATUS
213  OUTPUT_VARIABLE P
214  ERROR_VARIABLE ERROR
215  OUTPUT_STRIP_TRAILING_WHITESPACE
216  )
217  if (ERROR MATCHES "'import site' failed|ImportError: No module named site")
218  set (IMPORT_SITE_ERROR TRUE)
219  endif ()
220  set (ENV{PYTHONHOME} "${PYTHONHOME}")
221  endif ()
222  endif ()
223  if (STATUS EQUAL 0)
224  if (P MATCHES "__init__\\.pyc?$")
225  get_filename_component (P "${P}" PATH)
226  get_filename_component (P "${P}" PATH)
227  else ()
228  get_filename_component (P "${P}" PATH)
229  endif ()
230  set_property (CACHE ${CACHEVAR} PROPERTY VALUE "${P}")
231  return ()
232  elseif (IMPORT_SITE_ERROR)
233  message (WARNING "Import of site module failed when running Python interpreter ${ARGN_PYTHON_EXECUTABLE}"
234  " with and without -E option. Make sure that the Python interpreter is installed properly"
235  " and that the PYTHONHOME environment variable is either not set (recommended) or at"
236  " least set correctly for this Python installation. Maybe you need to enable this Python"
237  " version first somehow if more than one version of Python is installed on your system?"
238  " Otherwise, set PYTHON_EXECUTABLE to the right Python interpreter executable (python).")
239  endif ()
240  endif ()
241  # 3. search PYTHONPATH
242  if (NOT ARGN_NO_PYTHONPATH)
243  string (REPLACE ":" ";" PYTHONPATH "$ENV{PYTHONPATH}")
244  foreach (P ${PYTHONPATH}) # ignore empty entries
245  if (IS_ABSOLUTE "${P}")
246  if (EXISTS "${P}/${ARGN_NAME}.py" OR EXISTS "${P}/${ARGN_NAME}/__init__.py" OR
247  EXISTS "${P}/${ARGN_NAME}.pyc" OR EXISTS "${P}/${ARGN_NAME}/__init__.pyc")
248  set_property (CACHE ${CACHEVAR} PROPERTY VALUE "${P}")
249  return ()
250  endif ()
251  endif ()
252  endforeach ()
253  endif ()
254 endfunction ()
255 
256 # ----------------------------------------------------------------------------
257 # find Python modules
258 if (NOT PythonModules_FIND_COMPONENTS AND NOT PythonModules_FIND_OPTIONAL_COMPONENTS)
259  message (FATAL_ERROR "PythonModules: No (OPTIONAL_)COMPONENTS (i.e., Python module names) specified!")
260 endif ()
261 if (NOT DEFINED PythonModules_PYTHONPATH)
262  set (PythonModules_PYTHONPATH) # PYTHONPATH of all found modules
263 endif ()
264 # helper macro
265 macro (_PythonModules_find_python_modules ALL_FOUND)
266  set (${ALL_FOUND} TRUE)
267  foreach (_PythonModules_MODULE ${ARGN})
268  set (_PythonModules_VAR "PythonModules_${_PythonModules_MODULE}_PATH")
269  set (_PythonModules_TGT "PythonModules_${_PythonModules_MODULE}")
270  if (PythonModules_DIR)
271  basis_find_python_module (
272  ${_PythonModules_VAR}
273  NAME "${_PythonModules_MODULE}"
274  PATH "${PythonModules_DIR}"
275  NO_DEFAULT_PATH
276  )
277  else ()
278  basis_find_python_module (
279  ${_PythonModules_VAR}
280  NAME "${_PythonModules_MODULE}"
281  PATH "${PythonModules_DIR}"
282  )
283  endif ()
284  mark_as_advanced (${_PythonModules_VAR})
285  if (${_PythonModules_VAR} MATCHES "(^ *|NOTFOUND)$")
286  set (${ALL_FOUND} FALSE)
287  set (PythonModules_${_PythonModules_MODULE}_FOUND FALSE)
288  else ()
289  if (NOT TARGET ${_PythonModules_TGT})
290  add_library (${_PythonModules_TGT} UNKNOWN IMPORTED)
291  set_target_properties (
292  ${_PythonModules_TGT}
293  PROPERTIES
294  BASIS_TYPE "SCRIPT_LIBRARY"
295  IMPORTED_LOCATION "${${_PythonModules_VAR}}"
296  )
297  endif ()
298  set (PythonModules_${_PythonModules_MODULE}_FOUND TRUE)
299  list (APPEND PythonModules_PYTHONPATH "${${_PythonModules_VAR}}")
300  endif ()
301  endforeach ()
302 endmacro ()
303 # optional first, as PythonModules_FOUND shall be reset to TRUE afterwards
304 _PythonModules_find_python_modules (PythonModules_FOUND ${PythonModules_FIND_OPTIONAL_COMPONENTS})
305 _PythonModules_find_python_modules (PythonModules_FOUND ${PythonModules_FIND_COMPONENTS})
306 # remove duplicate paths in PYTHONPATH
307 if (PythonModules_PYTHONPATH)
308  list (REMOVE_DUPLICATES PythonModules_PYTHONPATH)
309 endif ()
310 
311 # ----------------------------------------------------------------------------
312 # handle standard QUIET and REQUIRED arguments
313 if (NOT PythonModules_FOUND)
314  # list of modules that were not found
315  set (_PythonModules_MISSING)
316  foreach (_PythonModules_MODULE ${PythonModules_FIND_COMPONENTS})
317  if (NOT PythonModules_${_PythonModules_MODULE}_FOUND)
318  list (APPEND _PythonModules_MISSING "${_PythonModules_MODULE}")
319  endif ()
320  endforeach ()
321  # hint on how to help finding the Python modules
322  set (_PythonModules_HINT)
323  if (PYTHON_EXECUTABLE)
324  set (_PythonModules_HINT "Check if executing ${PYTHON_EXECUTABLE} -c \"import <module>\" works")
325  else ()
326  set (_PythonModules_HINT "Set PYTHON_EXECUTABLE, e.g., by searching for Python interpreter first")
327  endif ()
328  if (_PythonModules_HINT)
329  set (_PythonModules_HINT "${_PythonModules_HINT} or set")
330  else ()
331  set (_PythonModules_HINT "Set")
332  endif ()
333  set (_PythonModules_HINT "${_PythonModules_HINT} the PythonModules_DIR variable. Alternatively,")
334  set (_PythonModules_HINT "${_PythonModules_HINT} set the PythonModules_<module>_PATH variable(s)")
335  set (_PythonModules_HINT "${_PythonModules_HINT} instead or the PYTHONPATH environment variable.")
336  set (_PythonModules_HINT "${_PythonModules_HINT} Unset PythonModules_DIR if you chose an alternative")
337  set (_PythonModules_HINT "${_PythonModules_HINT} option and before rerunning CMake again.")
338  # error message
339  string (REPLACE ";" ", " ${_PythonModules_MISSING} "${_PythonModules_MISSING}")
340  message (FATAL_ERROR "Could NOT find the following Python modules:\n${_PythonModules_MISSING}\n${_PythonModules_HINT}")
341 endif ()

351  FORCE
